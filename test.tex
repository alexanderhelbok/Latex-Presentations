	\begin{frame}[fragile]{Nested Structs}
\begin{myr}
struct person{
	char name[50];
	int age;
	float height;
}; 

struct Family{
	struct person;
	int number_Sisters;
	int number_Brothers;
}myFamily;

int main(){
	struct person Alex = {.age = 19, .height = 1.7};
	myFamily = {Alex, .number_Sister = 2};
	printf("Height = %f\n", Family.Alex.height);
}\end{myr}
	\end{frame}
